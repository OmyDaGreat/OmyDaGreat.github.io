StringBuilder & LinkedHashMap

System.out.println("Type the message to be shortened:");
String s = new Scanner(System.in).nextLine().toLowerCase();
System.out.println();
System.out.println("Algorithm 1");

/* Repeating characters */
int count1 = 0;
int count2 = 0;
StringBuilder one = new StringBuilder(s);
for (int i = s.length(); i > 1; i--) {
  String x = s.substring(i - 2, i - 1);
  String x2 = s.substring(i - 1, i);
  if (x.equals("a") || x.equals("e") || x.equals("i") || x.equals("o") || x.equals("u")) {
  } else if (x.equals(x2)) {
    count2++;
    one.deleteCharAt(i - 1);
  }
}

/* Vowels */
StringBuilder one2 = new StringBuilder(one.toString());
for (int i = one.length(); i > 0; i--) {
  String x = one.substring(i - 1, i);
  if (!(i == 1) && !one.substring(i - 2, i - 1).equals(" ") && (x.equals("a") || x.equals("e") || x.equals("i") || x.equals("o") || x.equals("u"))) {
    count1++;
    one2.deleteCharAt(i - 1);
  }
}

/* Done with Algorithm 1 */
System.out.println("Vowels removed: " + (count1));
System.out.println("Repeats removed: " + (count2));
System.out.println("Algorithm 1 message: " + one2);
System.out.println("Algorithm 1 characters saved: " + (count1 + count2));

/* Algorithm 2 */
System.out.println();
System.out.println("Algorithm 2");
LinkedHashMap <Character, Integer> map = new LinkedHashMap <> ();
String s2 = new String(s);
for (char c: s2.replaceAll("\\s", "").toCharArray()) {
  if (!map.containsKey(c)) {
    map.put(c, 1);
  } else {
    int freq = map.get(c);
    map.put(c, freq + 1);
  }
}
String output = "";
for (Map.Entry < Character, Integer > e: map.entrySet()) {
  output += e.getValue() + "" + e.getKey();
}
System.out.println("Unique characters found: " + map.size());
System.out.println("Algorithm 2 message: " + output);
System.out.println("Algorithm 2 characters saved: " + (s.length() - output.length()));
